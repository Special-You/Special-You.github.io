### 
```

var colors = [
                        ['#00eb0c', '#061232'],
                        ['#fff000', '#061437']
                    ]
                  例：  yData  = 2;
 var option = {
                        title: {
                            text: '控制方式',//AUTO AND MAN
                            left: 'center',
                            top: '0',
                            textStyle: {
                                fontSize: '14',
                                color: '#FFF',
                                fontWeight: 'normal',
                            },
                        },
                        
                        series: [{
                            name: 'AUTO',
                            type: 'pie',
                            clockWise: false,
                            radius: ['55%', '70%'],
                            startAngle: 90 - 360 * yData / 200,
                            itemStyle: {
                                normal: {
                                    color: colors[0][0],
                                    shadowColor: colors[0][0],
                                    shadowBlur: 0,
                                    label: {
                                        show: false
                                    },
                                    labelLine: {
                                        show: false
                                    },
                                }
                            },
                            hoverAnimation: false,
                            center: ['50%', '50%'],
                            data: [{
                                value: yData,
                                label: {
                                    normal: {
                                        formatter: function (params) {
                                            return 'AUTO\n'+params.value + '%';
                                        },
                                        position: 'center',
                                        show: true,
                                        textStyle: {
                                            fontSize: '28',
                                            fontWeight: 'bold',
                                            color: colors[0][0],
                                        }
                                    }
                                },
                            }, {
                                value: 100 - yData,
                                name: 'invisible',
                                itemStyle: {
                                    normal: {
                                        color: colors[0][1]
                                    },
                                    emphasis: {
                                        color: colors[0][1]
                                    }
                                }
                            }]
                        }]
                    };
                    
                    var echart = echarts.init(document.getElementById("wusen"));
                    echart.setOption(option);

```
