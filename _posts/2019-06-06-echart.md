#### x轴lable换行显示

```
xAxis: {
                axisLabel: {
                    interval: 0,
                    // rotate: 45,
                    margin: 5,
                    textStyle: {
                        color: "#222"
                    },
                    formatter:function(val){
                        var strs = val.split(''); //字符串数组
                        var str = '';
                        for(var i = 0, s; s = strs[i++];) {
                            //遍历字符串数组
                             str += s;
                             if(!(i % 8))
                                 str += '\n'; //按需要求余
                             }
                        return str
                    }

                },
                data: data

            },

```
